```javascript
import Database from 'better-sqlite3';
import bcrypt from 'bcryptjs';
import { fileURLToPath } from 'url';
import { dirname, join } from 'path';
import { mkdirSync, existsSync } from 'fs';

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);

const dataDir = join(__dirname, 'data');
if (!existsSync(dataDir)) {
  mkdirSync(dataDir, { recursive: true });
}

const dbPath = process.env.DB_PATH || join(dataDir, 'familywall.db');
const db = new Database(dbPath);
db.pragma('journal_mode = WAL');

// Schema initialization
db.exec(`
  CREATE TABLE IF NOT EXISTS family_members (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    created_at INTEGER DEFAULT (unixepoch()),
    updated_at INTEGER DEFAULT (unixepoch())
  );

  CREATE TABLE IF NOT EXISTS settings (
    key TEXT PRIMARY KEY,
    value TEXT NOT NULL,
    updated_at INTEGER DEFAULT (unixepoch())
  );

  CREATE INDEX IF NOT EXISTS idx_family_name ON family_members(name);
`);

// Seed default PIN (1234) if no PIN exists yet (per D-01)
const existingPin = db.prepare('SELECT value FROM settings WHERE key = ?').get('parental_pin');
if (!existingPin) {
  const defaultHash = bcrypt.hashSync('1234', 13);
  db.prepare(
    `INSERT INTO settings (key, value, updated_at) VALUES ('parental_pin', ?, unixepoch())`
  ).run(defaultHash);
}

export default db;
```

2. Create `server/routes/family.js` with these exact routes:

- **GET /api/family** — Returns all family members sorted alphabetically by name (per D-08). Schema response: array of `{id: integer, name: string, created_at: integer, updated_at: integer}`. SQL: `SELECT * FROM family_members ORDER BY name COLLATE NOCASE`.

- **POST /api/family** — Creates family member. Schema body: `{name: string, minLength: 1, maxLength: 100}` required. SQL: `INSERT INTO family_members (name) VALUES (?)`. Returns 201 with `{id, name, created_at, updated_at}`. Use `stmt.run(name.trim())` then fetch the created row by lastInsertRowid.

- **PUT /api/family/:id** — Updates family member name. Schema params: `{id: integer}`. Schema body: `{name: string, minLength: 1, maxLength: 100}` required. SQL: `UPDATE family_members SET name = ?, updated_at = unixepoch() WHERE id = ?`. Return updated row. If no row found (changes === 0), return 404 `{error: "Family member not found"}`.

- **DELETE /api/family/:id** — Deletes family member. Schema params: `{id: integer}`. SQL: `DELETE FROM family_members WHERE id = ?`. If no row found (changes === 0), return 404. Otherwise return 204.

All routes use `db.prepare()` with parameterized queries (never string concatenation).

3. Create `server/routes/auth.js` with these exact routes:

- **POST /api/auth/verify** — Verifies PIN. Schema body: `{pin: string, minLength: 4, maxLength: 6}` required. Schema response 200: `{valid: boolean}`. Fetch stored hash from settings table where key='parental_pin'. Use `await bcrypt.compare(pin, row.value)`. If no PIN row exists, return `{valid: false}`. Per D-03: unlimited retries, no lockout.

- **PUT /api/settings/pin** — Sets/changes PIN. Schema body: `{currentPin: string, newPin: string}` both required, minLength 4, maxLength 6. First verify currentPin against stored hash (per D-02: requires current PIN to change). If invalid, return 403 `{error: "Current PIN is incorrect"}`. If valid, hash newPin with `await bcrypt.hash(newPin, 13)` (per D-04: cost factor 13). Upsert into settings using `INSERT ... ON CONFLICT(key) DO UPDATE SET value = ?, updated_at = unixepoch()`. Return `{success: true}`.
